𖺯𖻁 𖺡𖺻𖻌𖻐𖺻: 𖺡𖺾𖻌𖻂𖺻
Titre en français: La vendeuse de lait
En langue beria, parlée au Tchad et au Soudan
Conte raconté par: Mahamat Saleh Khazali
Illustrations par: Banda Koussedi Richard
Traduit en langue beria par Souleyman Adam Ahmat
Écriture: scripte erfé
Première édition provisoire en 2006: 25 copies
Publié par: Bakan Assalam, BP 24, Abeche, TCHAD. E-mail: dsfaris@tchad.org

𖺩𖻑𖺼𖺻𖻓 𖻇𖻂, 𖺼𖺻𖻀𖻑 𖻊𖻁 𖺼𖺻𖻌𖻐𖺻 𖻇𖻂 𖻀𖺾𖻇𖻂 𖻌𖻑 𖻄𖺾𖻓 𖻄𖺾𖻓-𖻀𖻂𖻇𖻂́. 𖺡𖺾𖻌 𖻄𖺾𖻐𖻐𖻂𖺾́, 𖻊𖻁 𖻄𖻂𖻀𖻊 𖻇𖺻𖻌𖺾𖻇𖻂́.

𖺶 𖺼𖺾𖻌 𖻀𖺻𖻌𖺽𖻂 𖻌𖻑 𖻄𖺾𖻎𖻂𖻌𖻂𖻀𖻂́, 𖺼𖺾𖻌 𖻐𖻊𖺽𖻑 𖻄𖻂𖻀𖻊 𖻄𖺾𖻌𖻂 𖻑𖻓𖺻𖻌𖻂́, “𖺠𖻓 𖻑 𖻀𖺾𖻇𖻂 𖻌𖻑 𖻃𖻑𖻀𖺾̀, 𖻊𖻁 𖺾𖻀𖻂 𖻇𖺻𖻀𖻂𖻇, 𖻄𖻊𖺽𖻂𖻂 𖻇𖺻𖻀𖺾̀ 𖻄𖺾𖺽𖺾𖻀𖻂̀.”

𖺶 𖻄𖻊𖺽𖻂𖻂 𖺼𖻂𖻓𖺾 𖻌𖻑 𖻄𖺾𖺽𖺾-𖻀𖻂𖻇, 𖻀𖻑𖻇𖻑 𖺼𖺾𖻌𖻂́, “𖺶 𖻀𖻑𖻇𖻑 𖻄𖺾𖺼𖺾𖻇, 𖻄𖺻𖻅𖻇𖻂̀.”

𖺶 𖻄𖺻𖻅-𖻀𖻂𖻇𖻂́, 𖺼𖻑𖻌 𖻑𖻀𖻑𖻌𖺾𖻇𖻂́. 𖺶 𖻑𖻀𖻑𖻌𖺾-𖻀𖻂𖻇𖻂𖻈, 𖻐𖺾𖻀𖺾 𖻇𖺻𖻀𖻂́.

𖺶 𖻐𖺾𖻀𖺾 𖻇𖺻𖻀𖻂́, 𖻀𖻂𖻌𖻂𖻎 𖻐𖻊 𖻁𖻂𖻌𖻂 𖻇𖻆 𖻇𖻂 𖻇𖺻𖻀𖻂́. 𖺶 𖻁𖻂𖻌𖻂 𖻄𖻂 𖻇𖺻𖻀𖺾́, 𖺼𖻂𖻓𖺾 𖻌𖻑 𖻄𖻊𖻌𖻀𖻂𖻇 𖻆𖻂 𖻁𖻂𖻌𖻂.

𖺩𖻁𖺻𖻅𖺻𖻎, 𖻑 𖻆𖻂 𖻄𖻂 𖻆𖻂𖻇𖺻 𖻐𖺾𖻓𖻇𖻂́, 𖺼𖺾𖻌 𖻄𖻂𖻌𖺻𖻀𖻂 𖻌𖻑 𖺼𖻂𖻓𖺾 𖻐𖻊𖻒𖻌𖻂́.

𖺢 𖺾𖻀𖺾𖻌𖺾́, 𖺽𖺾𖻀𖺾𖻌𖺾́, 𖺽𖺾𖻀𖺾𖻌𖺾́ 𖻐𖻊𖻒𖻌𖻂́. 𖺩𖻁𖺻𖻅𖺻𖻎 𖺼𖺾𖻌 𖻀𖺻 𖻆𖻂𖻓𖺾𖻌𖺾́, 𖻐𖻊𖻒𖻌𖻂́.

𖺢𖺾𖻀𖻌𖺾́ 𖻐𖻊𖻒𖻌𖻂́, 𖻐𖻊𖻒𖻌𖻂́, 𖻐𖻊𖻒𖻌𖻂́, 𖻅𖺻𖻁𖺻𖺽𖻂 𖻎𖺾𖻀𖺾𖻌𖻂́ 𖻊𖻁 𖻄𖻂𖻀𖻊 𖻄𖻂 𖻂𖻌𖻂 𖻄𖺾𖺽𖺾𖻓𖺾́, 𖻐𖺻𖻓𖻇𖻂́.

𖺩𖻁𖺻𖻅𖺻𖻎, 𖻀𖺾𖻇𖻂 𖻄𖻂𖻀𖻊 𖻑𖻌𖻑-𖻀𖻂𖻇𖻂́, 𖺼𖻂𖻓𖺾 𖻌𖻑 𖻐𖻂𖺾 𖺾𖻇𖻇𖺾 𖻐𖺻 𖺼𖺻𖻌 𖻄𖺾𖻓𖻌𖻂́. 𖺡𖺾𖻌 𖺼𖺻𖻌𖻂 𖻄𖻂𖻀𖻊 𖻄𖺾𖻀𖺾𖻎𖺾𖻂́, 𖻐𖻊 𖻎𖻊𖻄𖻊 𖺾𖺼𖻂 𖻀𖻑 𖻄𖺾𖻎𖻐𖻂́, 𖻊𖻁 𖻄𖻂𖻀𖻊 𖻀𖻂𖻌𖺻 𖻄𖺻𖻇𖺻 𖺻𖻄𖻄𖻂𖺾́, 𖻄𖻂𖻇𖺻𖻓𖻊 𖻀𖺻𖻌𖺽𖻂 𖻌𖻑  𖻐𖺻𖻒𖺻𖻌𖻂́.
