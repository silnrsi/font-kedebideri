\centerline{\title 𖺯𖻁 𖺡𖺻𖻌𖻐𖺻: 𖺡𖺾𖻌𖻂𖺻}
\medskip

\par Titre en français: La vendeuse de lait
\par En langue beria, parlée au Tchad et au Soudan
\par Conte raconté par: Mahamat Saleh Khazali
\par Illustrations par: Banda Koussedi Richard
\par Traduit en langue beria par Souleyman Adam Ahmat
\par Écriture: scripte erfé
\medskip
\par Première édition provisoire en 2006: 25 copies
\medskip
\par Publié par: Bakan Assalam, BP 24, Abeche, TCHAD. E-mail: dsfaris@tchad.org

\par 𖺩𖻑𖺼𖺻𖻓 𖻇𖻂, 𖺼𖺻𖻀𖻑 𖻊𖻁 𖺼𖺻𖻌𖻐𖺻 𖻇𖻂 𖻀𖺾𖻇𖻂 𖻌𖻑 𖻄𖺾𖻓 𖻄𖺾𖻓-𖻀𖻂𖻇𖻂́. 𖺡𖺾𖻌 𖻄𖺾𖻐𖻐𖻂𖺾́, 𖻊𖻁 𖻄𖻂𖻀𖻊 𖻇𖺻𖻌𖺾𖻇𖻂́.
\par 𖺶 𖺼𖺾𖻌 𖻀𖺻𖻌𖺽𖻂 𖻌𖻑 𖻄𖺾𖻎𖻂𖻌𖻂𖻀𖻂́, 𖺼𖺾𖻌 𖻐𖻊𖺽𖻑 𖻄𖻂𖻀𖻊 𖻄𖺾𖻌𖻂 𖻑𖻓𖺻𖻌𖻂́, “𖺠𖻓 𖻑 𖻀𖺾𖻇𖻂 𖻌𖻑 𖻃𖻑𖻀𖺾̀, 𖻊𖻁 𖺾𖻀𖻂 𖻇𖺻𖻀𖻂𖻇, 𖻄𖻊𖺽𖻂𖻂 𖻇𖺻𖻀𖺾̀ 𖻄𖺾𖺽𖺾𖻀𖻂̀.”
\par 𖺶 𖻄𖻊𖺽𖻂𖻂 𖺼𖻂𖻓𖺾 𖻌𖻑 𖻄𖺾𖺽𖺾-𖻀𖻂𖻇, 𖻀𖻑𖻇𖻑 𖺼𖺾𖻌𖻂́, “𖺶 𖻀𖻑𖻇𖻑 𖻄𖺾𖺼𖺾𖻇, 𖻄𖺻𖻅𖻇𖻂̀.”
\par 𖺶 𖻄𖺻𖻅-𖻀𖻂𖻇𖻂́, 𖺼𖻑𖻌 𖻑𖻀𖻑𖻌𖺾𖻇𖻂́. 𖺶 𖻑𖻀𖻑𖻌𖺾-𖻀𖻂𖻇𖻂𖻈, 𖻐𖺾𖻀𖺾 𖻇𖺻𖻀𖻂́.
\par 𖺶 𖻐𖺾𖻀𖺾 𖻇𖺻𖻀𖻂́, 𖻀𖻂𖻌𖻂𖻎 𖻐𖻊 𖻁𖻂𖻌𖻂 𖻇𖻆 𖻇𖻂 𖻇𖺻𖻀𖻂́. 𖺶 𖻁𖻂𖻌𖻂 𖻄𖻂 𖻇𖺻𖻀𖺾́, 𖺼𖻂𖻓𖺾 𖻌𖻑 𖻄𖻊𖻌𖻀𖻂𖻇 𖻆𖻂 𖻁𖻂𖻌𖻂.
\par 𖺩𖻁𖺻𖻅𖺻𖻎, 𖻑 𖻆𖻂 𖻄𖻂 𖻆𖻂𖻇𖺻 𖻐𖺾𖻓𖻇𖻂́, 𖺼𖺾𖻌 𖻄𖻂𖻌𖺻𖻀𖻂 𖻌𖻑 𖺼𖻂𖻓𖺾 𖻐𖻊𖻒𖻌𖻂́.
\par 𖺢 𖺾𖻀𖺾𖻌𖺾́, 𖺽𖺾𖻀𖺾𖻌𖺾́, 𖺽𖺾𖻀𖺾𖻌𖺾́ 𖻐𖻊𖻒𖻌𖻂́. 𖺩𖻁𖺻𖻅𖺻𖻎 𖺼𖺾𖻌 𖻀𖺻 𖻆𖻂𖻓𖺾𖻌𖺾́, 𖻐𖻊𖻒𖻌𖻂́.
\par 𖺢𖺾𖻀𖻌𖺾́ 𖻐𖻊𖻒𖻌𖻂́, 𖻐𖻊𖻒𖻌𖻂́, 𖻐𖻊𖻒𖻌𖻂́, 𖻅𖺻𖻁𖺻𖺽𖻂 𖻎𖺾𖻀𖺾𖻌𖻂́ 𖻊𖻁 𖻄𖻂𖻀𖻊 𖻄𖻂 𖻂𖻌𖻂 𖻄𖺾𖺽𖺾𖻓𖺾́, 𖻐𖺻𖻓𖻇𖻂́.
\par 𖺩𖻁𖺻𖻅𖺻𖻎, 𖻀𖺾𖻇𖻂 𖻄𖻂𖻀𖻊 𖻑𖻌𖻑-𖻀𖻂𖻇𖻂́, 𖺼𖻂𖻓𖺾 𖻌𖻑 𖻐𖻂𖺾 𖺾𖻇𖻇𖺾 𖻐𖺻 𖺼𖺻𖻌 𖻄𖺾𖻓𖻌𖻂́. 𖺡𖺾𖻌 𖺼𖺻𖻌𖻂 𖻄𖻂𖻀𖻊 𖻄𖺾𖻀𖺾𖻎𖺾𖻂́, 𖻐𖻊 𖻎𖻊𖻄𖻊 𖺾𖺼𖻂 𖻀𖻑 𖻄𖺾𖻎𖻐𖻂́, 𖻊𖻁 𖻄𖻂𖻀𖻊 𖻀𖻂𖻌𖺻 𖻄𖺻𖻇𖺻 𖺻𖻄𖻄𖻂𖺾́, 𖻄𖻂𖻇𖺻𖻓𖻊 𖻀𖺻𖻌𖺽𖻂 𖻌𖻑  𖻐𖺻𖻒𖺻𖻌𖻂́.
